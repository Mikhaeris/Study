uses GraphABC;

begin
  SetWindowSize(500, 500);
  Coordinate.SetOrigin(-20, -80);
  
  var w := 99;
  var h := 49;
  
  for var y := 0 to 10 do
    for var x := 0 to 10 do
      begin
        var xOffset := (y mod 2) * (w div 2);
        var xPos := x * w - xOffset;
        var yPos := y * h;
        
        Rectangle(xPos, yPos, xPos + 40, yPos + 60);
        Rectangle(xPos + 39, yPos + 10, xPos + 100, yPos +50);
      end;
end.
